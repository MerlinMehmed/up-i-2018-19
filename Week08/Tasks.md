#### 1 задача
Какъв е резултатът от изпълнението на програмата? 

```
#include <iostream.h>
int main()
{
	int a = 3, b = 7, x;
	int *pa = &a, *pb = &b;
	cout << *pa << " " << *pb << endl;
	x = *pa;
	*pa = *pb;
	*pb = x;
	cout << *pa << " " << *pb << endl;
	return 0;
}
```

#### 2 задача
Какъв е резултатът от изпълнението на програмата? 

```
#include <iostream.h>
int main()
{
	int a = 5, b = 4;
	int *pa = &a, *pb = &b, *p;
	cout << *pa << " " << *pb << endl;
	p = pa;
	pa = pb;
	pb = p;
	cout << *pa << " " << *pb << endl;
	return 0;
}
```
